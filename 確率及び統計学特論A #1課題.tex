\begin{tcolorbox}
	$(\Omega, \F, \mathcal{P})$を確率空間とする．このとき
\begin{equation} \tag{Ex}
	\mathcal{P}(\emptyset)= 0
\end{equation}
	を示せ．
\end{tcolorbox}

\begin{answer}
	背理法で示す．$\mathcal{P}(\emptyset)= a\ne 0$と仮定し，$A_i= \begin{cases}\Omega\ (i= 1)\\ 0\ (i\ne 0)\end{cases}$とする．
\begin{equation*}
	1
	= \mathcal{P}(A_1)
	= \mathcal{P}\left(\sum_{i= 1}^\infty A_i\right)\\
	= \sum_{i= 1}^\infty \mathcal{P}(A_i)\\
	= 1+ \sum_{i= 2}^\infty a
\end{equation*}
	ここで$\sum_{i= 2}^\infty a\to \infty$となるが，左辺の値と一致せず矛盾．
	よって$a= 0$であり，$\mathcal{P}(\emptyset)= 0$
\end{answer}
